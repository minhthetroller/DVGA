    class App {
        - store *Store
        - sessions *Manager
        - registry *Registry
        - chain *Chain
        - difficulty *Difficulty
        - handler *Handler
        + New(dbPath, templatesDir, staticDir) *App
        + ListenAndServe(addr string) error
    }

    class Difficulty {
        <<enumeration>>
        Easy
        Medium
        Hard
        + String() string
        + ParseDifficulty(s string) Difficulty
    }

    class Registry {
        - factories map[string]ModuleFactory
        + Register(id string, f ModuleFactory)
        + Build(id string, d Difficulty) VulnModule
        + All(d Difficulty) []VulnModule
        + IDs() []string
        + Categories(d Difficulty) map
    }

    class Chain {
        - decorators []MiddlewareDecorator
        + Use(m MiddlewareDecorator) *Chain
        + Apply(VulnModule) VulnModule
    }

    class Store {
        - db *gorm.DB
        + NewStore(dbPath string) *Store
        + DB() *gorm.DB
        + AutoMigrate() error
        + Seed() error
        + Reset() error
    }

    class Manager {
        <<session>>
        - sessions map[string]*Session
        - loginAttempts map[string]*AttemptTracker
        - resetAttempts map[string]*AttemptTracker
        + Create(userID int, username, role string) string
        + Get(token string) *Session
        + Destroy(token string)
        + RecordLoginAttempt(key string) *AttemptTracker
        + RecordResetAttempt(key string) *AttemptTracker
        + ClearLoginAttempts(key string)
        + ClearResetAttempts(key string)
    }

    class Session {
        + UserID int
        + Username string
        + Role string
        + CreatedAt time.Time
    }

    class AttemptTracker {
        + Count int
        + LastFail time.Time
        + LockUntil time.Time
    }

    class Handler {
        <<ui>>
        - renderer *TemplateRenderer
        - registry *Registry
        - chain *Chain
        - store *Store
        - sessions *Manager
        - difficulty *Difficulty
        + Routes() chi.Router
    }

    class TemplateRenderer {
        - templates *template.Template
        + NewRenderer(dir string) *TemplateRenderer
        + Render(w io.Writer, name string, data PageData)
    }

    class ModuleFactory {
        <<interface>>
        + Create(d Difficulty) VulnModule
    }

    class MiddlewareDecorator {
        <<interface>>
        + Wrap(inner VulnModule) VulnModule
    }

    class VulnModule {
        <<interface>>
        + ServeHTTP(w ResponseWriter, r *Request)
        + Meta() ModuleMeta
    }

    class ModuleMeta {
        + ID string
        + Name string
        + Description string
        + Category string
        + Difficulty Difficulty
        + References []string
    }
